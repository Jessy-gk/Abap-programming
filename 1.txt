SELECTION-SCREEN BEGIN OF BLOCK A WITH FRAME TITLE TEXT-001.
  SELECT-OPTIONS : S_MATNR FOR MARA-MATNR,
  S_MTART FOR MARA-MTART.
    SELECTION-SCREEN END OF BLOCK A.

    TYPES: BEGIN OF TY_MARA,
            MATNR TYPE MARA-MATNR,
            MTART TYPE MARA-MTART,
            MBRSH TYPE MARA-MBRSH,
            MATKL TYPE MARA-MATKL,
          END OF TY_MARA.


PARAMETERS P_DIS RADIOBUTTON GROUP G.
PARAMETERS P_DOWN RADIOBUTTON GROUP G.

    DATA : IT_MARA TYPE STANDARD TABLE OF TY_MARA, WA_MARA TYPE TY_MARA.
    SELECT MATNR MTART MBRSH MATKL FROM MARA
      INTO TABLE IT_MARA WHERE MATNR IN S_MATNR AND MTART IN S_MTART.

      IF IT_MARA IS INITIAL.
        MESSAGE 'NO DATA FOUND' TYPE 'E'.
        ELSE.
          LOOP AT IT_MARA INTO WA_MARA.
            WRITE :/ WA_MARA-MATNR,WA_MARA-MTART,WA_MARA-MBRSH,WA_MARA-MATKL.
            ENDLOOP.
            ENDIF.
